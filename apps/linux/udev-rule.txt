# /etc/udev/rules.d/99-input.rules
# Allows members of the 'input' group to access /dev/uinput for synthetic key emission.
#
# Install this rule:
#   sudo cp udev-rule.txt /etc/udev/rules.d/99-input.rules
#   sudo udevadm control --reload
#   sudo udevadm trigger
#
# Add your user to the input group:
#   sudo usermod -a -G input $USER
#   # Log out and back in, or:
#   newgrp input
#
# Verify:
#   ls -l /dev/uinput
#   groups $USER

KERNEL=="uinput", MODE="0660", GROUP="input"
